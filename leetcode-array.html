53. Maximum Subarray

易错点, 如果所有的数都是负数, 可能得出最大值为零的错误.

class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        
        maxsum, currsum = nums[0], 0
        
        if(max(nums)<0):
            return max(nums) #这一步关键
        
        for i in range(len(nums)):
            currsum += nums[i]
            if currsum < 0:
                currsum = 0
            maxsum = max(maxsum, currsum)
                    
        return maxsum

我给出的解答, 细微差别!

class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        max_sub, curr_sub = nums[0], 0
        for num in nums:
            if curr_sub + num > 0:
                curr_sub += num
                max_sub = max(max_sub, curr_sub)
            else:
                max_sub = max(max_sub, num)
                curr_sub = 0
        
        return max_sub

88. Merge Sorted Array

此题关键是处理边界以及停止 while 的条件:

class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        while m > 0 and n > 0:
            if nums1[m-1] > nums2[n-1]:
                nums1[m+n-1] = nums1[m-1]
                m -= 1
            else:
                nums1[m+n-1] = nums2[n-1]
                n -= 1
        
        if n > 0:
            nums1[:n] = nums2[:n]

如果使用 while m > 0 or n > 0 在边界时会出问题.

448. Find All Numbers Disappeared in an Array

下面这种方法会超时

class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        return [x for x in range(1,len(nums)+1) if x not in nums]

另外两种方法

class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        
        return list(set(range(1, len(nums)+1))-set(nums))
class Solution(object):
    def findDisappearedNumbers(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        # For each number i in nums,
        # we mark the number that i points as negative.
        # Then we filter the list, get all the indexes
        # who points to a positive number
        for i in range(len(nums)):
            index = abs(nums[i]) - 1
            nums[index] = - abs(nums[index])

        return [i + 1 for i in range(len(nums)) if nums[i] > 0]

643. Maximum Average Subarray I

class Solution:
    def findMaxAverage(self, nums: List[int], k: int) -> float:
        sums = sum(nums[0:k])
        res = sums / k
        for i in range(1,len(nums) - k + 1):
            sums = sums - nums[i-1] + nums[i+k-1]     #这一步是关键, 如果每次都是直接求和会超时!           
            res = max(sums/ k, res)
        return res

704. Binary Search 典型二分法

class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:  # 此处分三步处理,而不是简单的 if-else!if-else!
                return mid
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        return -1

746 Min Cost Climbing Stairs

如果用 recursion 解决, 通常不是最佳答案, 需要重新考虑解法, 类似地还有斐波那契数列. 容易超时!

class Solution:
    def minCostClimbingStairs(self, cost: List[int]) -> int:
        if len(cost) < 3:
            return min(cost)
        elif len(cost) == 3:
            return min(cost[0]+cost[2], cost[1])
        else:
            return min(self.minCostClimbingStairs(cost[:len(cost)-1]) + cost[len(cost)-1], 
                self.minCostClimbingStairs(cost[:len(cost)-2]) + cost[len(cost)-2])

DP - Dynamic Programming

class Solution:
    def minCostClimbingStairs(self, cost: List[int]) -> int:
        dp1, dp2 = 0, 0
        for i in range(len(cost)):
            dp1, dp2 = dp2, cost[i] + min(dp1, dp2)
            
        return min(dp1, dp2)

922. Sort Array By Parity II

class Solution:
    def sortArrayByParityII(self, nums: List[int]) -> List[int]:
        odd, even = [], []
        for num in nums:
            if num % 2: odd.append(num)
            else: even.append(num)
        res =[0 for _ in nums]
        for i in range(len(odd)):
            res[2*i] = even[i]
            res[2*i+1] = odd[i]
        
        return res

双指针 in-place

class Solution:
    def sortArrayByParityII(self, nums: List[int]) -> List[int]:
        i, j = 0, 1
        while i < len(nums) and j < len(nums):
            if not (nums[i] - i) % 2: i += 2
            elif not (nums[j] - j) % 2: j += 2
            else: 
                nums[i], nums[j] = nums[j], nums[i]
                i += 2
                j += 2
        return nums

class Solution:
    def validMountainArray(self, arr: List[int]) -> bool:
        if len(arr) < 3:
            return False
        else:
            left, right = 0, len(arr) -1
            while left < right and arr[left] < arr[left + 1]:
                left += 1
            while left < right and arr[right] < arr[right - 1]:
                right -= 1
            return (left == right) and (left != 0) and (right != len(arr) -1)

class Solution:
    def validMountainArray(self, arr: List[int]) -> bool:
        i, j, n = 0, len(arr) - 1, len(arr)
        while i + 1 < n and arr[i] < arr[i + 1]: i += 1
        while j > 0 and arr[j - 1] > arr[j]: j -= 1
        return 0 < i == j < n - 1

1475. Final Prices With a Special Discount in a Shop

这里用 stack 用得应该不对!

class Solution:
    def finalPrices(self, prices: List[int]) -> List[int]:
        if len(prices) == 1: return prices
        i = len(prices) - 1
        res = prices[:]
        stack = []
        while i >= 0:
            for num in stack[::-1]:
                if num <= prices[i]: 
                    res[i] = prices[i] - num
                    break #这个不要漏掉!
            while stack and stack[-1] >= prices[i]:
                stack.pop()
            stack.append(prices[i])
            i -= 1
        return res

懵逼脸! 直接用两个 for loop 也没慢多少…

class Solution:
    def finalPrices(self, prices: List[int]) -> List[int]:
        lp=len(prices)
        ret=[]
        for i in range(lp):
            for j in range(i+1,lp):
                if prices[j]<=prices[i]:
                    ret+=[prices[i]-prices[j]]
                    break
                if j==lp-1:
                    ret+=[prices[i]]
        ret.append(prices[-1])
        return ret
class Solution:
    def finalPrices(self, prices: List[int]) -> List[int]:
        stack = []
        for i, a in enumerate(prices):
            while stack and prices[stack[-1]] >= a:
                prices[stack.pop()] -= a
            stack.append(i)
        return prices
