LeetCode 473. Matchsticks to Square 06/15/2021

发表于2021-06-16|更新于2021-06-16|PythonLeetcode
|阅读量:|评论数:

06/15/2021

class Solution:
    def makesquare(self, matchsticks: List[int]) -> bool:
        def dfs(nums, pos, target):
            if pos == len(nums): return True
            for i in range(4):
                if nums[pos] <= target[i]:
                    target[i] = target[i] - nums[pos]
                    if dfs(nums, pos + 1, target): return True
                    target[i] = target[i] + nums[pos]
            return False
        
        matchsticks.sort(reverse = True)
        if len(matchsticks) < 4 or sum(matchsticks) % 4 or matchsticks[0] > sum(matchsticks)// 4:
            return False
        target = [sum(matchsticks)// 4] * 4
        return dfs(matchsticks, 0, target)
